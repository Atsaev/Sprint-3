class Employee:
    vacation_days = 28

    # инициализируем
    def __init__(self, first_name, second_name, gender):

        self.first_name = first_name
        self.second_name = second_name
        self.gender = gender

    def get_vacations_details(self, consume_vacations):

        self.consume_vacations = consume_vacations
        self.days_remaining = Employee.vacation_days - consume_vacations
        return f'Осталось {self.days_remaining} дней отпуска'

    def result(self):

        return f'Имя: {self.first_name}, Фамилия: {self.second_name}, Пол: {self.gender}, Отпускных дней в году: {Employee.vacation_days}.'


# Создайте экземпляры класса Employee с различными значениями атрибутов.
employee1 = Employee(first_name='Роберт', second_name='Крузо', gender='м')
employee2 = Employee(first_name='Анна', second_name='Смит', gender='ж')
used_days_emp1 = 7
used_days_emp2 = 13

# Допишите код для вывода информации о сотрудниках.
print(employee1.result())
print(employee1.get_vacations_details(used_days_emp1))
print(employee2.result())
print(employee2.get_vacations_details(used_days_emp2))
